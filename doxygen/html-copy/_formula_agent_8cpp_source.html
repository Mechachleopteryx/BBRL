46 int FormulaAgent::getAction(int xt) const throw (AgentException)
47 {
48  unsigned int nU = getMDP()->getNbActions();
49 
50 
51  // Compute the Q-function with formula 'f'
52  std::vector<double> Qxt;
53  for (unsigned int u = 0; u < nU; ++u)
54  {
55  std::vector<double> v;
56  for (unsigned int i = 0; i < varList.size(); ++i)
57  {
58  if (varList[i]) { v.push_back((*varList[i])(xt, u)); }
59  else { v.push_back(0.0); }
60  }
61 
62 
63  try { Qxt.push_back((*f)(v)); }
64  catch (FormulaException& e) { throw AgentException(e.what()); }
65  }
66 
67 
68  // Select the action maximizing the computed Q-function
69  vector<vector<double>::const_iterator> maxList;
70  maxList = utils::search::max<vector<double> >(Qxt.begin(), Qxt.end());
71 
72  assert(!maxList.empty());
73 
74  int maxI = RandomGen::randIntRange_Uniform(0, (maxList.size() - 1));
75  return (maxList[maxI] - Qxt.begin());
76 }
