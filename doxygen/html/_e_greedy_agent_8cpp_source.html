50 int EGreedyAgent::getAction(int xt) const throw (AgentException)
51 {
52  assert((xt >= 0) && (xt < (int) nX));
53  assert(Q.size() == (getMDP()->getNbStates() * getMDP()->getNbActions()));
54 
55 
56  double f = RandomGen::rand01_Uniform();
57 
58  // 'random' case
59  if (f <= epsilon)
60  return RandomGen::randIntRange_Uniform(0, (nU - 1));
61 
62  // 'greedy' case
63  else
64  {
65  vector<vector<double>::const_iterator> maxList;
66  maxList = utils::search::max<vector<double> >(
67  (Q.begin() + nU*xt), (Q.begin() + nU*xt + nU));
68 
69  assert(!maxList.empty());
70 
71  int maxI = RandomGen::randIntRange_Uniform(0, (maxList.size() - 1));
72  return (maxList[maxI] - (Q.begin() + nU*xt));
73  }
74 }
