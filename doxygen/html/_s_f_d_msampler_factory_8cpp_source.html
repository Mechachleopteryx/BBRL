5 /* Model: S --> V --> P --> N
6  * where S ~ P(S = k) size of set of outcomes
7  * V ~ Uniform in sets of size S
8  * P ~ Restricted Dirichlet to V, with alpha prior
9  * N ~ Multinomial(P)
10  *
11  * Sample Method 1:
12  *
13  * For each s,a:
14  * Given data N, compute integrated P directly in closed-form (See Friedman Singer paper)
15  * and sample from it.
16  *
17  * Sample Method 2:
18  *
19  * For each s,a:
20  * Given data N, sample P (exact inference) to sample MDP models.
21  * First sample P(S | N) then P(V | N, S), then P(P | V, N)
22  */

169 Sampler* SFDMsamplerFactory::getTransitionSampler(const uint* counts, uint, uint, uint S){
170 
171  double* P = new double[S];
172 
173  uint K0 = 0;
174  uint N = 0;
175  for(uint ll=0;ll<S;++ll){
176  if(counts[ll] > 0){
177  K0++;
178  N += counts[ll];
179  }
180  }
181  //Sample Method 1
182  uint K0M = std::max(1,(int)K0);
183  //Compute C(D,L)
184  double CDL;
185  if(N == 0)
186  CDL = 0;
187  else if(N < S)
188  CDL = CDLpre[K0-1][N];
189  else{
190  CDL = 0;
191  double sumMk = 0;
192  for(uint i=K0;i<=S;++i){
193  double logmk;
194  logmk = log(Pk[i-1])+sumLog[i-1]+lgam[(i-1)*S];
195  if(N<S)
196  logmk -= lgam[(i-1)*S+N];
197  else
198  logmk -= lgamma(i*alpha+N);
199  if(i!=K0)
200  logmk -= sumLog[i-K0-1];
201  double mk = exp(logmk);
202  CDL += (K0M*alpha+N)*mk/(i*alpha+N);
203  sumMk += mk;
204  }
205  if(sumMk > 0)
206  CDL = CDL / sumMk;
207  else
208  CDL = 1;
209  }
210  double leftover = (1-CDL)/(S-K0);
211  double CDLK0alpN = CDL/(K0M*alpha+N);
212  double sum = 0;
213  for(uint ll=0;ll<S;++ll){
214  if(counts[ll] == 0)
215  P[ll] = leftover;
216  else
217  P[ll] = CDLK0alpN*(alpha+counts[ll]);
218  sum += P[ll];
219 
220  }
221  SFDMTransitionSampler* sampler = new SFDMTransitionSampler(P,S);
222  return (Sampler*) sampler;
223 }

231  //Update P(k | D)

234  double mk, logmk;

237  if(CSum[ind]<S)
238  logmk -= lgam[(i-1)*S+CSum[ind]];
239  else
240  logmk -= lgamma(i*alpha+CSum[ind]);
241  if(i!=K0)
242  logmk -= sumLog[i-K0-1];
