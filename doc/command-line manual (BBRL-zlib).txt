NAME
     BBRL-zlib -- A simple compression/uncompression program based on zlib (by Castronovo Michael)



USAGE OVERVIEW
     --help
     
     --compress               (-z)
          --compression_level (-cl) [possible values: Z_NO_COMPRESSION, Z_BEST_COMPRESSION, Z_BEST_SPEED, Z_DEFAULT_COMPRESSION]
          "input file"
     
     --uncompress             (-d)
          "output file".zz



USAGE EXAMPLES
     --- Compress ---
          
          ./BBRL-zlib --compress --compression_level "compression level" "file to compress"
          
                         or
          
          ./BBRL-zlib -z  -cl "compression level" "file to compress"


     --- Uncompress ---
          
          ./BBRL-zlib --uncompress "file to uncompress"
     
                         or
     
          ./BBRL-zlib -d "file to uncompress"
